func
 
BenchmarkSuccessfulAuth
(
b
 
*
testing
.
B
)
 
{

	
const
 
ldapToken
 
string
 
=
 
"LDAP_TOKEN"

	
f
 
:=
 
frisby
.
Create
(
"Check performance of valid authorization token"
)
.
Get
(
tests
.
API_URL
)

	
if
 
os
.
Getenv
(
ldapToken
)
 
==
 
""
 
{

		
f
.
AddError
(
"Please provide LDAP_TOKEN env variable!"
)

	
}
 
else
 
{

		
f
.
SetHeader
(
"Authorization"
,
 
"Bearer "
+
os
.
Getenv
(
ldapToken
)
)

		
for
 
i
 
:=
 
0
;
 
i
 
<
 
b
.
N
;
 
i
++
 
{

			
f
.
Send
(
)

		
}

	
}
